Table Orders {
    id int [primary key]
    drink_id int
    order_date_time datetime
}

Table Drinks {
    id int [primary key]
    name varchar(100)
    enabled bool
    img image
    toppings varchar(255)
}

Table DrinkIngredients {
    drink_id int [primary key]
    ingredient_name string [primary key]
    ml int
}

Table Ingredients {
    name varchar(100) [primary key]
    remaining_liquid int
}

Table Pumps {
  slot int [primary key]
  ingredient_name string
  active boolean
}

Ref: Orders.drink_id > Drinks.id
Ref: Drinks.id > DrinkIngredients.drink_id
Ref: Ingredients.name > DrinkIngredients.ingredient_name
Ref: Ingredients.name - Pumps.ingredient_name

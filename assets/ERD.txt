Table order_log {
    id int [primary key]
    drink_id int
    order_date_time datetime
}

Table drink {
    id int [primary key]
    name varchar(100)
    enabled bool
    img image
    toppings varchar(255)
}

Table drink_ingredient {
    drink_id int [primary key]
    ingredient_name string [primary key]
    ml int
}

Table ingredient {
    name varchar(100) [primary key]
    slot int [unique]
    remaining_liquid int
}

Ref: order_log.drink_id > drink.id
Ref: drink.id > drink_ingredient.drink_id
Ref: ingredient.name > drink_ingredient.ingredient_name

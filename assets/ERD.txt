Table Drink {
  id int [primary key]
  name varchar(100)
  enabled bool
  img varchar(1000)
  toppings varchar(255)
}

Table Ingredient {
  drink_id int
  ingredient_name varchar(100)
  ml int

  // Composite primary key:
  indexes {
    (drink_id, ingredient_name) [pk]
  }
}

Table IngredientInBottle {
  name varchar(100) [primary key]
  remaining_ml int
  max_ml int
}

Table Pump {
  slot int [primary key]
  ingredient_name varchar(100) [unique]  // Enforces 1:1 with IngredientInBottle
  active bool
}

Table Order {
  id int [primary key]
  drink_id int
  order_date_time datetime
}

// Relationships
Ref: Ingredient.drink_id > Drink.id
Ref: Ingredient.ingredient_name > IngredientInBottle.name
Ref: Order.drink_id > Drink.id
Ref: Pump.ingredient_name > IngredientInBottle.name

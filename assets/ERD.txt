Table order {
    id int [primary key]
    drink_id int
    order_date_time datetime
}

Table drink {
    id int [primary key]
    name varchar(100)
    enabled bool
    img image
    toppings varchar(255)
}

Table ingredient {
    drink_id int [primary key]
    drink_name varchar(100)
    required_ml int
}

Table bottle {
    id int [primary key]
    drink_name varchar(100) [unique]
    remaining_ml int
}

Table Pump {
  slot int [primary key]
  bottle_id varchar(100)
  active boolean
}

Ref: Pump.bottle_id - bottle.id
Ref: order.drink_id > drink.id
Ref: drink.id < ingredient.drink_id
Ref: ingredient.drink_name - bottle.drink_name
